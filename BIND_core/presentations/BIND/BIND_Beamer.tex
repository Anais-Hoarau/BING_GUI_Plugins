	\frame{
		\frametitle{Moteur d'accès aux données}
		\begin{itemize}
			\item BIND prend en charge l'accès aux données via son interface
			\item On ne touche \textbf{jamais} aux données sans passer par le framework
			\item L'implémentation de l'interface logicielle prend en charge les fonctions de bas niveau
		\end{itemize}
	}


	\frame{
		\frametitle{Sauvegarde des trips}
		\begin{itemize}
			\item BIND définit une interface pour les formats de données
			\begin{itemize}
				\item Tout format respectant l'interface est dit BINDable
			\end{itemize}
			\item Une implémentation de cette interface existe : \textbf{SQLite}
			\item On peut en imaginer d'autres
		\end{itemize}
